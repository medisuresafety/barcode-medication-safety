<script src="https://cdnjs.cloudflare.com/ajax/libs/quagga/0.12.1/quagga.min.js"></script>
<script>
function startScanner() {
    document.getElementById('result').innerText = "Scanning...";
    Quagga.init({
        inputStream: { name: "Live", type: "LiveStream", target: document.querySelector('#scanner-container'), constraints: { facingMode: "environment" } },
        decoder: { readers: ["code_128_reader", "ean_reader"] }
    }, function(err) {
        if (err) { console.error(err); return; }
        Quagga.start();
    });

    Quagga.onDetected(function(data) {
        const code = data.codeResult.code;
        document.getElementById('result').innerText = `Scanned code: ${code}`;
        Quagga.stop();
    });
}
</script>
